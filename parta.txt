from diffusers import StableDiffusionPipeline 
import torch 
 
# Load model 
pipe = StableDiffusionPipeline.from_pretrained( 
    "runwayml/stable-diffusion-v1-5", 
     torch_dtype=torch.float16 
).to("cpu")   # use "cpu" if no GPU 
 
# Input prompt 
prompt = input("Enter prompt: ") 
 
# Generate image 
image = pipe(prompt).images[0] 
 
# Save and show 
image.save("output.png") 
image.show() 
 
print("Image saved as output.png")

from transformers import pipeline
qa_pipeline= pipeline("question-answering")

def ask_question(context,question):
    result=qa_pipeline(question=question,context=context)
    return result['answer']
context = """ 
AI is a simmulation of human brain.
Ai is used widely in today in multiple fields such as hospital,agriculture,robotics,
its mainly focus on automate the human work without much interfare of human.


""" 
question=input("enter your question")
answer=ask_question(context,question)
print("answer:",answer)

from transformers import pipeline

generator = pipeline("text-generation", model="gpt2")

prompt = """
You are an expert AI researcher.

Write a short article on "Applications of Generative AI in Healthcare".

Include:
1. Introduction
2. Key applications (at least 3)
3. Benefits
4. Conclusion

Keep the explanation clear and concise.
"""

result = generator(prompt,
                   max_length=100,
                   num_return_sequences=1)[0]['generated_text']

print("\n=== Engineered Prompt ===\n")
print(prompt)

print("\n=== Generated Article ===\n")
print(result)
